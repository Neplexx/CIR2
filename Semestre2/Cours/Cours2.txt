fonction echange2(entre : x,y ; sors : x,y):

	y = x - y;
	x = x - y;

fonction factorielleRec(entre: entier; sors : entier):
	si (entier <= 1) : 
		renvoie 1;
	sinon :
		renvoie factorielleRec(entier - 1) * entier

fonction factorielle(entre: entier; sors: entier):
	for ( i = entier - 1 ; i > 0 ; i-=1):
		entier = entier * i
	renvoie entier

fonction somme(entre : entier; sors : entier): 
	si (entier <=1) : 
		renvoie 1;
	sinon : 
		renvoie somme(entier-1) + entier

		produitScalaire = produitScalaire + V1[i] * V2[i];

fonction produitEuclidien(entrées : a,b ; sorties : r,q):
	r = a;
	tant que (r - b > 0):
		r = r - b;
		q = q + 1;
	renvoie r,q
